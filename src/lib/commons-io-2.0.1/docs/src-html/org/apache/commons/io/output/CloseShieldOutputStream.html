017    package org.apache.commons.io.output;
018    
019    import java.io.OutputStream;
020    
021    /**
022     * Proxy stream that prevents the underlying output stream from being closed.
023     * <p>
024     * This class is typically used in cases where an output stream needs to be
025     * passed to a component that wants to explicitly close the stream even if
026     * other components would still use the stream for output.
027     *
028     * @version $Id: CloseShieldOutputStream.java 659817 2008-05-24 13:23:10Z niallp $
029     * @since Commons IO 1.4
030     */
031    public class CloseShieldOutputStream extends ProxyOutputStream {
032    
033        /**
034         * Creates a proxy that shields the given output stream from being
035         * closed.
036         *
037         * @param out underlying output stream
038         */
039        public CloseShieldOutputStream(OutputStream out) {
040            super(out);
041        }
042    
043        /**
044         * Replaces the underlying output stream with a {@link ClosedOutputStream}
045         * sentinel. The original output stream will remain open, but this proxy
046         * will appear closed.
047         */
048        @Override
049        public void close() {
050            out = new ClosedOutputStream();
051        }
052    
053    }
