017    package org.apache.commons.io.filefilter;
018    
019    import java.io.File;
020    import java.io.FileFilter;
021    import java.io.FilenameFilter;
022    
023    /**
024     * An interface which brings the FileFilter and FilenameFilter 
025     * interfaces together.
026     * 
027     * @since Commons IO 1.0
028     * @version $Revision: 471628 $ $Date: 2006-11-06 04:06:45 +0000 (Mon, 06 Nov 2006) $
029     * 
030     * @author Stephen Colebourne
031     */
032    public interface IOFileFilter extends FileFilter, FilenameFilter {
033    
034        /**
035         * Checks to see if the File should be accepted by this filter.
036         * <p>
037         * Defined in {@link java.io.FileFilter}.
038         * 
039         * @param file  the File to check
040         * @return true if this file matches the test
041         */
042        public boolean accept(File file);
043    
044        /**
045         * Checks to see if the File should be accepted by this filter.
046         * <p>
047         * Defined in {@link java.io.FilenameFilter}.
048         * 
049         * @param dir  the directory File to check
050         * @param name  the filename within the directory to check
051         * @return true if this file matches the test
052         */
053        public boolean accept(File dir, String name);
054        
055    }
